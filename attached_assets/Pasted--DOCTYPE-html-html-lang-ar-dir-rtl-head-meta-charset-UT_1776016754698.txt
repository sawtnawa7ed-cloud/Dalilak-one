<script>
    let db;
    // إنشاء قاعدة بيانات IndexedDB للملفات الكبيرة (فيديو/صور)
    const request = indexedDB.open("SawtnaProDB", 1);
    
    request.onupgradeneeded = e => {
        db = e.target.result;
        db.createObjectStore("content", { keyPath: "id", autoIncrement: true });
        db.createObjectStore("profile", { keyPath: "id" });
    };

    request.onsuccess = e => { db = e.target.result; loadProfile(); renderPosts(); };

    function toggleAdmin() {
        const p = document.getElementById('adminPanel');
        p.style.display = p.style.display === "block" ? "none" : "block";
    }

    // حفظ بيانات المبادرة (الشعار، من نحن، الاتصال)
    function saveProfile() {
        const profile = {
            id: "site_info",
            about: document.getElementById('aboutInp').value,
            phone: document.getElementById('phoneInp').value,
            email: document.getElementById('emailInp').value,
            logo: null
        };

        const file = document.getElementById('newLogo').files[0];
        if (file) {
            const reader = new FileReader();
            reader.onload = e => {
                profile.logo = e.target.result;
                storeProfile(profile);
            };
            reader.readAsDataURL(file);
        } else {
            storeProfile(profile);
        }
    }

    function storeProfile(p) {
        const tx = db.transaction("profile", "readwrite");
        tx.objectStore("profile").put(p);
        tx.oncomplete = () => { alert("تم حفظ البيانات!"); location.reload(); };
    }

    function loadProfile() {
        const tx = db.transaction("profile", "readonly");
        tx.objectStore("profile").get("site_info").onsuccess = e => {
            const p = e.target.result;
            if (p) {
                document.getElementById('aboutTxt').innerText = p.about || "صوتنا واحد";
                document.getElementById('footerContacts').innerHTML = `
                    <span class="contact-info">📞 ${p.phone}</span>
                    <span class="contact-info">📧 ${p.email}</span>
                `;
                if(p.logo) document.getElementById('siteLogo').src = p.logo;
                
                // ملء الخانات في الإدارة
                document.getElementById('aboutInp').value = p.about;
                document.getElementById('phoneInp').value = p.phone;
                document.getElementById('emailInp').value = p.email;
            }
        };
    }

    // نشر المنشورات الحرة (فيديو طويل، صور، نصوص)
    function handlePost() {
        const t = document.getElementById('postT').value;
        const c = document.getElementById('postC').value;
        const file = document.getElementById('postM').files[0];

        if (!t && !c && !file) return;

        const post = { t, c, file: file || null, type: file ? (file.type.startsWith('video') ? 'vid' : 'img') : null, date: new Date().toLocaleString('ar-LB') };
        
        const tx = db.transaction("content", "readwrite");
        tx.objectStore("content").add(post);
        tx.oncomplete = () => { location.reload(); };
    }

    function renderPosts() {
        const feed = document.getElementById('feed');
        const tx = db.transaction("content", "readonly");
        tx.objectStore("content").openCursor(null, 'prev').onsuccess = e => {
            const cursor = e.target.result;
            if (cursor) {
                const p = cursor.value;
                const div = document.createElement('div');
                div.className = 'post';
                
                let media = "";
                if (p.file) {
                    const url = URL.createObjectURL(p.file);
                    media = p.type === 'vid' ? `<video controls src="${url}"></video>` : `<img src="${url}">`;
                }

                div.innerHTML = `
                    <small style="color:var(--gold)">${p.date}</small>
                    ${p.t ? `<h2>${p.t}</h2>` : ''}
                    ${p.c ? `<p>${p.c}</p>` : ''}
                    ${media}
                    <br><button onclick="deletePost(${p.id})" style="background:#ff4444; color:white; border:none; padding:5px 10px; border-radius:5px; margin-top:15px; cursor:pointer;">حذف</button>
                `;
                feed.appendChild(div);
                cursor.continue();
            }
        };
    }

    function deletePost(id) {
        if(confirm("هل تريد حذف هذا التقرير؟")) {
            const tx = db.transaction("content", "readwrite");
            tx.objectStore("content").delete(id);
            tx.oncomplete = () => location.reload();
        }
    }
</script>
